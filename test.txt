var PAGES = ['home','assess','results','history','stats','graphql'];
var currentFormData = null;
var rowCount = 0;
var chartObjs = {};

function goTo(name) {
  for (var i = 0; i < PAGES.length; i++) {
    var el = document.getElementById(PAGES[i]);
    if (el) el.style.display = 'none';
  }
  var btns = document.querySelectorAll('.nav-links button');
  for (var j = 0; j < btns.length; j++) btns[j].classList.remove('active');
  var show = document.getElementById(name);
  if (show) show.style.display = 'block';
  var nb = document.getElementById('nav-' + name);
  if (nb) nb.classList.add('active');
  window.scrollTo(0, 0);
  if (name === 'history') loadHistory();
  if (name === 'stats') loadStats();
}

var BASE = 'http://localhost:5000/api';

async function apiPost(url, data) {
  var r = await fetch(BASE + url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  });
  return r.json();
}

async function apiGet(url) {
  var r = await fetch(BASE + url);
  return r.json();
}

async function apiDel(url) {
  var r = await fetch(
